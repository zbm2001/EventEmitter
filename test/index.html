Quick start

// 创建一个事件触发器的实例
function Foo(){}
EventEmitter.inherito(Foo);
var foo = new Foo();

foo instanceof EventEmitter; //=> true

// 注册事件侦听器：
foo.on('hint', function(e){
	console.log(e.type)
});

// 触发事件:
foo.emit('hint');

// 删除事件侦听器:
foo.off('hint');
